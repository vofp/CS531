Adjacent([x, y], [u, v]) <-> x = u & (Succ(y,v) | Succ(v,y)) | y = v & (Succ(x, u) | Succ(u, y)).

At(Agent,x,z) & Breeze(z)	-> Breezy(x).
At(Agent,x,z) & -Breeze(z)	-> -Breezy(x).

At(Agent,x,z) & Stench(z)	-> Smelly(x).
At(Agent,x,z) & -Stench(z)	-> -Smelly(x).

-Breezy(x) & Adjacent(x,y) -> -Pit(y).
-Smelly(x) & Adjacent(x,y) -> -(At(Wumpus,y,z)).

all x (exists y (Adjacent(x,y) & Pit(y)) <-> Breezy(x)).
all x (exists y (Adjacent(x,y) & (At(Wumpus,y,z))) <-> Smelly(x)).
